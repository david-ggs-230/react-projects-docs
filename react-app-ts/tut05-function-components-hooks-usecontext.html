5.5.3. React Hooks - useContext

useContext is used to access values from a React Context without needing to pass props manually at every level of the component tree. Context is one of the important concept in React. It provides the ability to pass a information from the parent component to all its children to any nested level without passing the information through props in each level. Context will make the code more readable and simple to understand. Context can be used to store information which does not change or have minimal change. Some of the use cases of context are as follows

  • Application configuration

  • Current authenticated user information

  • Current user setting

  • Language setting

  • Theme / Design configuration by application / users

Context usage through hook

  • Creating a new context

    // Create a context
const ValueContext = React.createContext("default value");
// Create a Context with Multiple Objects
const ThemeContext = React.createContext({
   color: 'black',
   backgroundColor: 'white'
})
// Create a Context with Objects and functions
const ThemeContext = React.createContext<ThemeType>({
  theme: { color: "black", backgroundColor: "white" },
  setTheme: () => {},
});

  • Setting context provider in the root component

    <ThemeContext.Provider value={{
   color: 'white',
   backgroundColor: 'green'
}}>
    <children components.../>
</ThemeContext.Provider>

  • Setting context consumer in the component where we need the context information

    import ThemeContext from "ThemeContext";
const theme = useContext(ThemContext)

  • Accessing context information and using it in render method

    let theme = useContext(ThemeContext)
return (
   <div style={{
      color: theme.color,
      backgroundColor: theme.backgroundColor }}>
         Hello World
   </div>
)

Updating context: Updating the context will rerender all the child component. React provides an option to update the context by using both useState and useContext hook.

  • In the root component, use useState hook to manage the theme information

    const [theme, setTheme] = useState({...})
<ThemeContext.Provider value={{ theme, setTheme }}>
    <children components.../>
</ThemeContext.Provider>

  • In the component where we need the context information, use useContext and state update function

    import ThemeContext from "ThemeContext";
let { theme, setTheme } = useContext(ThemeContext)
const handleClick=(color)=>{
  setTheme({color: color});
}

5.5.3.1.1. Create a React Project

    yarn create vite tut05-function-components-hooks-usecontext --template react-ts

5.5.3.2. Component - useContext

5.5.3.2.1. Function Components - the useContext Hook

  • Define a module to hold the context objects.

    Listing 5.54 src/CreateContextObjects.tsx
     1import React, { Dispatch, SetStateAction } from "react";
 2
 3// Create a context
 4const ValueContext = React.createContext("default value");
 5
 6type ThemeType = {
 7  theme: { color: string; backgroundColor: string };
 8  setTheme: Dispatch<
 9    SetStateAction<{ color: string; backgroundColor: string }>
10  >;
11};
12// Create a Context
13const ThemeContext = React.createContext<ThemeType>({
14  theme: { color: "black", backgroundColor: "white" },
15  setTheme: () => {},
16});
17export { ValueContext, ThemeContext };

  • Define a function component to use the context objects.

    Listing 5.55 src/ComponentUseContextValue.tsx
     1import { useContext } from "react";
 2import { ValueContext } from "./CreateContextObjects";
 3
 4import "./list-style.css";
 5
 6const ComponentUseContextValue = () => {
 7  // Access context value
 8  const value = useContext(ValueContext);
 9  return (
10    <>
11      <h5
12        className="blue-color"
13        style={{ marginTop: "20px", marginBottom: "0px" }}
14      >
15        <div>useContext with an initial value</div>
16      </h5>
17      <div style={{ marginTop: "0px" }}>
18        initial context value: <span className="red-color">{value}</span>
19      </div>
20    </>
21  );
22};
23
24export default ComponentUseContextValue;

  • Define a function component to update the context objects.

    Listing 5.56 src/ComponentUpdateContextValue.tsx
     1import { useContext, useRef } from "react";
 2import { ThemeContext } from "./CreateContextObjects";
 3import "./list-style.css";
 4
 5const ComponentUpdateContextValue = () => {
 6  // Access context value
 7  const { theme, setTheme } = useContext(ThemeContext);
 8  const blueBtnBorderRef = useRef("white");
 9  const redBtnBorderRef = useRef("white");
10  const handleClick = (color: string) => {
11    if (color === "red") {
12      redBtnBorderRef.current = "lightcoral";
13      blueBtnBorderRef.current = "white";
14    } else if (color === "blue") {
15      blueBtnBorderRef.current = "lightblue";
16      redBtnBorderRef.current = "white";
17    }
18    setTheme({ color: color, backgroundColor: color });
19  };
20  return (
21    <>
22      <h5
23        className="blue-color"
24        style={{ marginTop: "20px", marginBottom: "0px" }}
25      >
26        <div>useContext with useState to update context value</div>
27      </h5>
28      <div style={{ marginTop: "0px" }}>
29        Context Color:{" "}
30        <span style={{ color: theme.color }}>{theme.color} </span>
31      </div>
32      <div style={{ marginTop: "0px" }}>
33        <button
34          style={{ backgroundColor: redBtnBorderRef.current }}
35          onClick={() => handleClick("red")}
36        >
37          Red
38        </button>
39        <button
40          style={{
41            marginLeft: "10px",
42            backgroundColor: blueBtnBorderRef.current,
43          }}
44          onClick={() => handleClick("blue")}
45        >
46          Blue
47        </button>
48      </div>
49    </>
50  );
51};
52
53export default ComponentUpdateContextValue;

  • Define a function component with the useContext hook.

    Listing 5.57 src/ComponentuseContext.tsx
     1import { useState } from "react";
 2import { ThemeContext, ValueContext } from "./CreateContextObjects";
 3import ComponentUpdateContextValue from "./ComponentUpdateContextValue";
 4import ComponentUseContextValue from "./ComponentUseContextValue";
 5
 6import "./list-style.css";
 7
 8const ComponentUseContext = () => {
 9  const [theme, setTheme] = useState({
10    color: "green",
11    backgroundColor: "lightgreen",
12  });
13
14  return (
15    <>
16      <ValueContext.Provider value="Hello, World!">
17        <ComponentUseContextValue />
18      </ValueContext.Provider>
19      <ThemeContext.Provider value={{ theme: theme, setTheme: setTheme }}>
20        <ComponentUpdateContextValue />
21      </ThemeContext.Provider>
22    </>
23  );
24};
25
26export default ComponentUseContext;
